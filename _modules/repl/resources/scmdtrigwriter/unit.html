from eudplib import *

from ...utils import EPDConstString
from . import getWriter

@EUDFunc
def SCMDWriteUnit_var(unit):
    if EUDIf()(unit >= 233):
        getWriter().write_decimal(unit)
    if EUDElseIf()(unit == 228):
        getWriter().write_decimal(unit)
    if EUDElse()():
        getWriter().write_strepd(SCMDDefaultUnitArray[unit])
    EUDEndIf()
