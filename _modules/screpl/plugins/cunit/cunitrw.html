[docs]    @staticmethod
    def data(offset, write_f, name, comments = None):
        off_epd, off = divmod(offset, 4)
        name = EPDConstString(name)
        write_f = EUDFuncPtr(2, 0)(write_f)
        if comments is None:
            comments = EPD(0)
        else:
            comments = EPDConstString(comments)

        entry = CUnitMemberEntry.initialize_with(
            off_epd,
            off,
            write_f,
            name,
            comments,
            1
        )
        return entry
